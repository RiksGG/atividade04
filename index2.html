// C++ code

int contador = 0;
            
int counter;
            
void setup()
  {
pinMode(12, OUTPUT);
pinMode(4, OUTPUT);
pinMode(8, OUTPUT);
            
    contador = 3;
  }
            
void loop()
  {
for (counter = 0; counter < 2; ++counter) {
digitalWrite(12, HIGH);
delay(7000); // Wait for 7000 millisecond(s)
digitalWrite(12, LOW);
digitalWrite(4, HIGH);
delay(5000); // Wait for 5000 millisecond(s)
digitalWrite(4, LOW);
digitalWrite(8, HIGH);
delay(2000); // Wait for 2000 millisecond(s)
digitalWrite(8, LOW);
  }
while (contador > 1) {
digitalWrite(8, HIGH);
delay(1000); // Wait for 1000 millisecond(s)
digitalWrite(8, LOW);
delay(1000); // Wait for 1000 millisecond(s)
  }
}
